    export class Cache {

  constructor (props) {
    this._images = {}
    this._timestamp = Date.now()
    this._imagesContext = require.context('assets', false, /\.(png|jpe?g|svg|gif)$/)
  }

  get images () {
    return this._images
  }

  hasImage (name) {
    return (this.images[name] !== undefined)
  }

  cacheImage (id) {
    const name = `./${id}`
    const data = this._imagesContext(name, true)
    const timestamp = Date.now()
    this._images[id] = { data, timestamp }
  }

  image (name) {
    if (!this.hasImage(name)) {
      this.cacheImage(name)
    }

    return this.images[name]
  }
}

const cache = new Cache()
export default cache
